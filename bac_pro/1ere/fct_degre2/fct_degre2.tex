\section{Définition}

\begin{mydef}
	Une fonction du second degré est une fonction du type :
	\kw{\begin{align*}
		f(x) = ax^2 + bx +c
	\end{align*}}
	Où $a$, $b$ et $c$ sont des nombres quelconques, avec \kw{$a \neq 0$}.
\end{mydef}


\begin{myact}{1}
		
	\actOne{\disp}	
	
\end{myact}	

\section{Courbe représentative et variations}

\begin{myact}{2}
	\actTwo
\end{myact}

\begin{myprops}
	\begin{itemize}
		\item La courbe représentative d'une fonction du second degré est une \kw{parabole}.
		\item Le sens de la parabole dépend du signe de $a$
			\begin{itemize}
				\item Si \kw{$a > 0$} alors la parabole a un minimum (\kw{tête en bas}).
				\item Si \kw{$a < 0$} alors la parabole a un maximum (\kw{tête en haut}).
			\end{itemize}
		\item Le \kw{sommet de la courbe} (minimum ou maximum) est le point de coordonnées \kw{$( \frac{-b}{2a} ; f(\frac{-b}{2a} )$}.
	\end{itemize}


\begin{multicols}{2}

Cas où $a$ est positif

\begin{tikzpicture}[scale=0.8]
\tkzTabInit{$x$/1,$f(x)$/2}{$-\infty$, $\frac{-b}{2a}$, $+\infty$}
\tkzTabVar{+/,-/$f(\frac{-b}{2a})$,+/}
\end{tikzpicture}

Cas où $a$ est négatif

\begin{tikzpicture}[scale=0.8]
\tkzTabInit{$x$/1,$f(x)$/2}{$-\infty$, $\frac{-b}{2a}$, $+\infty$}
\tkzTabVar{-/,+/$f(\frac{-b}{2a})$,-/}
\end{tikzpicture}

\end{multicols}

\end{myprops}

		\item $g(x)=-2x^2+6x$ avec $-0,5 \leq x \leq 3,5$
	\end{itemize}
