\section{Calculer (8 points)}


\begin{questions}
	\question[4] Calculer les expressions suivantes en détaillant tous les calculs :
	
	\begin{enumerate}
		\item $A = 62 - 30 - 7 + 20$
		
		\item $B = (50 - (13 + 1) \times 2) - 6$
		
		\item $C= 225 - ((15 + 7) \times 10 - 2)$
		
		\item $D= (10 \times \num{3.2} - (79 - 71)) \div 6$
	\end{enumerate}


	\begin{solution}
		
		\begin{multicols}{2}
			
		\begin{eqnarray*}
			A &=& 62 - 30 - 7 + 20 \\
			A &=& 32 - 7 + 20 \\
			A &=& 25 + 20 \\
			A &=& 45
		\end{eqnarray*}
		
		\begin{eqnarray*}
			B &=& (50 - (13 + 1) \times 2) - 6 \\
			B &=& (50 - 14 \times 2) - 6  \\
			B &=& (50 - 28) - 6  \\
			B &=& 22 - 6  \\
			B &=& 16
		\end{eqnarray*}
		
		\begin{eqnarray*}
			C &=& 225 - ((15 + 7) \times 10 - 2)\\
			C &=& 225 - (22 \times 10 - 2) \\
			C &=& 225 - (220 - 2) \\
			C &=& 225 - 218  \\
			C &=& 7
		\end{eqnarray*}
		
		\begin{eqnarray*}
			D &=& (10 \times \num{3.2} - (79 - 71)) \div 6\\
			D &=& (10 \times \num{3.2} - 8) \div 6 \\
			D &=& (32 - 8) \div 6 \\
			D &=& 24 \div 6  \\
			D &=& 4
		\end{eqnarray*}
		\end{multicols}
	\end{solution}

	\question[1] Placer des parenthèses dans $A = 62 - 30 - 7 + 20$ pour trouver 59.
	\begin{solution}
		\begin{eqnarray*}
			A &=& 62 - (30 - 7) + 20 \\
			A &=& 62 - 23 + 20 \\
			A &=& 39 + 20 \\
			A &=& 59
		\end{eqnarray*}
	\end{solution}
	
	\question[1] Placer des parenthèses dans $A = 62 - 30 - 7 + 20$ pour trouver 19.
	\begin{solution}
		\begin{eqnarray*}
			A &=& 62 - (30 - 7 + 20) \\
			A &=& 62 - (23 + 20) \\
			A &=& 62 - 43\\
			A &=& 19
		\end{eqnarray*}
	\end{solution}
\end{questions}
