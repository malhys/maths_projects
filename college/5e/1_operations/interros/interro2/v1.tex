	\section{Calculer}
	Calculer les expressions suivantes en détaillant tous les calculs:
	\begin{questions}
		
		\question[3]  $A =  44 + 37 - 15 + 28$
		
		\fillwithdottedlines{6cm}
		{\LARGE \begin{solution}
			\begin{eqnarray*}
				A &=& 44 + 37 - 15 + 28 \\
				A &=& 81 - 15 + 28 \\
				A &=& 66 + 28 \\
				A &=& 94
			\end{eqnarray*}
		\end{solution}}
		
		
		\question[3]  $B = (7 + 5) \times (8 - 2)$
		
		\fillwithdottedlines{6cm}
		{\LARGE \begin{solution}
			\begin{eqnarray*}
			B &=& (7 + 5) \times (8 - 2)\\
			B &=& 12 \times 6 \\			
			B &=& 72
			\end{eqnarray*}
		\end{solution}}
