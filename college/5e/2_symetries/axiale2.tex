\begin{mydef}
	Deux figures sont \kw{\hspace*{4cm} par rapport à \hspace*{3cm} $(d)$} si elles se superposent quand on plie le long de cette droite. 
	La droite $(d)$ est appelée \kw{\hspace*{5cm}}.

\end{mydef}
