\section{Périmètre}

\subsection{Définition}

\begin{mydef}
	Le \kw{périmètre} d'une figure fermée est la longueur de son contour.
\end{mydef}

\begin{mymeth}
	Le périmètre d'un polygone est égal à la somme des longueurs de ses côtés.
\end{mymeth}

\subsection{Périmètre d'un cercle}

\begin{mydef}
	Le périmètre (ou la circonférence) d'un cercle est égal au produit de son diamètre par le nombre $\pi$.
\end{mydef}

\begin{myprop}
	\begin{multicols}{2}
		Pour calculer le périmètre d'un cercle \mbox{\begin{cursive}C\end{cursive}} on utilise une des formules suivantes :
		\begin{itemize}
			\item $ 2 \times R \times \pi$
			\item $ D \times \pi$
		\end{itemize}
	
	
		\begin{center}
			\includegraphics[scale=0.65]{cercle}
		\end{center}
	\end{multicols}
\end{myprop}


\section{Aire}

\begin{mydef}
	L'\kw{aire} d'une figure est la mesure de sa surface.
\end{mydef}

\section{Unités de mesure}

\begin{mydef}
	L'unité usuelle de longueur est le \kw{mètre} ($m$). L'unité usuelle d'aire est le \kw{mètre carré} ($m^2$). On utilise souvent des multiples et des sous-multiples.	
\end{mydef}
