	}{%
		$a$, $b$, $c$ et $d$  sont des nombres entiers avec $b \neq 0$ et $ d \neq 0$.
		
		$\dfrac{a}{b} = \dfrac{c}{d}$ signifie que $a \times d = b \times c$.
	}
	
	
	 
\end{myprop}
