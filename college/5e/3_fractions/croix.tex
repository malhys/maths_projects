\begin{myprop}
	$a$, $b$, $c$ et $d$  sont des nombres entiers avec $b \neq 0$ et $ d \neq 0$.
	
	$\frac{a}{b} = \frac{c}{d}$ signifie que $a \times d = b \times c$.
	
	 
\end{myprop}
