\subsection{Définition}

\begin{mydefs}
	
	\iftoggle{eleve}{%
		\vspace*{0.2cm}
		\hrulefill
		
		\vspace*{0.2cm}
		\hrulefill
		
		\vspace*{0.2cm}
		\hrulefill
		
		\begin{itemize}
			\item $a$ est \hrulefill
			\item $a$ est \hrulefill
			\item $b$ est \hrulefill
		\end{itemize}
	}{%
		Quand le reste de la division euclidienne du nombre $a$ par le nombre $b$, différent de zéro, est égal à zéro, on dit que : 
		
		\begin{itemize}
			\item $a$ est \kw{divisible} par $b$;
			\item $a$ est un \kw{multiple} de $b$;
			\item $b$ est un \kw{diviseur} de $a$.
		\end{itemize}
	}
	
	
\end{mydefs}

\begin{myex}
	\begin{multicols}{2}
		\begin{center}
			\opidiv{936}{24}
		\end{center}
		
		$24 \times 39 + 0 = 936$
	\end{multicols}

	\iftoggle{eleve}{%
		\hrulefill
		
		\vspace*{0.2cm}
		\hrulefill
	}{%
		936 est divisible par 24 ; 936 est un multiple de 24 ; 24 est un diviseur de 936.
	}

	
\end{myex}

\subsection{Critères de divisibilité}

\begin{myprops}
	
	\iftoggle{eleve}{%
		\begin{itemize}
			\item Un nombre entier est divisible par 2 si \hrulefill
			
			\vspace*{0.2cm}
			\hrulefill
			
			\item Un nombre entier est divisible par 5 si \hrulefill
			
			\vspace*{0.2cm}
			\hrulefill
			
			\item Un nombre entier est divisible par 10 si \hrulefill
			
			\vspace*{0.2cm}
			\hrulefill
			
			\item Un nombre entier est divisible par 3 si \hrulefill
			
			\vspace*{0.2cm}
			\hrulefill
			
			\item Un nombre entier est divisible par 9 si \hrulefill
			
			\vspace*{0.2cm}
			\hrulefill
			
			\item Un nombre entier est divisible par 4 si \hrulefill
			
			\vspace*{0.2cm}
			\hrulefill
			
			\vspace*{0.2cm}
			\hrulefill
		\end{itemize}
	}{%
		\begin{itemize}
			\item \item Un nombre entier est divisible par 2 si \kw{il est pair} (son chiffre des unités est , 2, 4, 6 ou 8);
			\item Un nombre entier est divisible par 5 si son \kw{chiffre des unités est 0 ou 5};
			\item Un nombre entier est divisible par 10 si son \kw{chiffre des unités est 0};	
			
			\item Un nombre entier est divisible par 3 si \kw{la somme de ses chiffres est divisible par 3};
			\item Un nombre entier est divisible par 9 si \kw{la somme de ses chiffres est divisible par 9};
			
			\item Un nombre entier est divisible par 4 si \kw{le nombre formé par ses chiffres des dizaines et des unités est divisible par 4}.
		\end{itemize}
	}
	
\end{myprops}

\begin{myexs}
	\iftoggle{eleve}{%
		\begin{itemize}
			\item $1250$ est divisible par : 
			\item $726$ est divisible par : 
			\item $1024$ est divisible par :
			\item $342$ est divisible par : 
			
		\end{itemize}
	}{%
		\begin{itemize}
			\item $1250$ est divisible par : 2; 5 et 10. 
			\item $726$ est divisible par : 2 et 3.
			\item $1024$ est divisible par : 2 et 4.
			\item $342$ est divisible par : 2; 3 et 9.
			
		\end{itemize}
	}
	
\end{myexs}
