			\item $\dfrac{16}{4}$
			\item $\dfrac{99}{54}$

	\end{itemize}

	\begin{center}
		$\dfrac{25}{4} \quad  ; \quad \dfrac{3}{7} \quad ; \quad \dfrac{3}{2} \quad ; \quad \dfrac{5}{14} \quad ; \quad 1 \quad ; \quad \dfrac{12}{8} $
