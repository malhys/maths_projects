\begin{mydef}
	Un point placé sur une demi-droite graduée est repéré par un nombre, son \kw{abscisse}.
\end{mydef}

\begin{myex}
	\begin{center}
		\includegraphics[scale=0.5]{img/axe}
	\end{center}
	
	\begin{itemize}
		\item L'abscisse du point A est :
		\item L'abscisse du point B est :
		\item L'abscisse du point C est : 500;
		\item L'abscisse du point D est \num{1100}.
	\end{itemize}
\end{myex}



\begin{mydefs}
	\begin{itemize}
		\item \kw{Comparer} des nombres, c'est dire si 		
		\iftoggle{eleve}{%
			\iftoggle{dys}{%
				un est plus \hspace{3cm} ou plus \hspace{3cm} que l'autre ou s'ils sont \hspace{3cm}.
			}{%
				\vspace*{0.75cm}
			}
		}{%
			un est plus petit ou plus grand que l'autre ou s'ils sont égaux.
		}
		\item Classer des nombres par \kw{ordre croissant} c'est 
		\iftoggle{eleve}{%
			\iftoggle{dys}{%
				les ranger du plus \hspace*{3cm} au plus \hspace*{3cm}.
			}{%
				\vspace*{0.75cm}
			}
		}{%
			les ranger du plus petit au plus grand.
		}
		
		\item Classer des nombres par \kw{ordre décroissant} c'est 
		\iftoggle{eleve}{%
			\iftoggle{dys}{%
				les ranger du plus \hspace*{3cm} au plus \hspace*{3cm}.
			}{%
				\vspace*{0.75cm}
			}
		}{%
			les ranger du plus grand au plus petit.
		}
		
		\item \kw{Encadrer} un nombre, c'est trouver 
		\iftoggle{eleve}{%
			\iftoggle{dys}{%
				un nombre plus \hspace*{3cm} \textbf{et} un nombre plus \hspace*{3cm} que lui.
			}{%
				\vspace*{0.75cm}
			}
		}{%
			un nombre plus petit \textbf{et} un nombre plus grand que lui.
		}
		
		
		
		\item \kw{Intercaler} un nombre entre deux autres, c'est trouver
		
		\iftoggle{eleve}{%
			\iftoggle{dys}{%
				un troisième nombre compris entre \\.
			}{%
				\vspace*{0.75cm}
			}
		}{%
			un troisième nombre compris entre les deux premiers.
		} 
	\end{itemize}
\end{mydefs}
