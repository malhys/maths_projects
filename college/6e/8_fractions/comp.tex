\begin{myprops}
	\begin{itemize}
		
		\iftoggle{eleve}{%
			\item Si deux \hrulefill
			
			\vspace*{0.3cm}
			\hrulefill
			
			\vspace*{0.3cm}
			\hrulefill
			
			
			
			\item Pour comparer \hrulefill
			
			\vspace*{0.3cm}
			\hrulefill
			
			\vspace*{0.3cm}
			\hrulefill
		}{%
			\item Si deux fractions ont le même dénominateur, la plus grande est celle qui a \kw{le plus grand numérateur}.
			
			\item Pour comparer deux fractions qui ont un dénominateur différent, on les écrit avec le même dénominateur.
		}
		
	\end{itemize}
\end{myprops}

\begin{myexs}
\begin{itemize}
	\item On veut comparer $\dfrac{3}{7}$ et $\dfrac{5}{7}$ : 
	
	
	\iftoggle{eleve}{%
		\vspace*{0.2cm} \hrulefill
	}{%
		3 < 5; donc $\dfrac{3}{7} < \dfrac{5}{7}$
	}
	
	
	\vspace*{0.5cm}
	
	\item On veut comparer $\dfrac{7}{3}$ et $\dfrac{13}{6}$ : 
	
	\iftoggle{eleve}{%
		\begin{itemize}
			\item \vspace*{0.2cm} \hrulefill
			
			\item \vspace*{0.2cm} \hrulefill
			
			
			\vspace*{1cm}
		\end{itemize}
	}{%
		\begin{itemize}
			\item On peut écrire $\dfrac{7}{3} = \dfrac{7 \times 2}{3 \times 2} = \dfrac{14}{6}$
			
			\item On a 13 < 14 donc $\dfrac{13}{6} < \dfrac{14}{6}$ en conclusion $\dfrac{13}{6} < \dfrac{7}{3}$
		\end{itemize}
	}
	
\end{itemize}
\end{myexs}


\begin{myprop}
	$a$ et $b$ désignent deux nombres ($b > 0$).
	
	\iftoggle{eleve}{%
		\begin{itemize}
			\item Si $a > b$ \hrulefill
			\item Si $a < b$ \hrulefill
			\item Si $a = b$ \hrulefill
		\end{itemize}
	}{%
		\begin{multicols}{3}
			\begin{itemize}
				\item Si $a > b$ alors $\dfrac{a}{b}$ > 1
				\item Si $a < b$ alors $\dfrac{a}{b}$ < 1
				\item Si $a = b$ alors $\dfrac{a}{b}$ = 1
			\end{itemize}
		\end{multicols}	
	}
\end{myprop}

\begin{myex}
	On veut comparer 1 ; $\dfrac{3}{4}$ et $\dfrac{15}{12}$ :
	
	\iftoggle{eleve}{%
			\begin{itemize}
				\item \vspace*{0.2cm} \hrulefill
				
				\item \vspace*{0.2cm} \hrulefill
				
				
				\vspace*{1cm}
			\end{itemize}
	}{%
		\begin{itemize}
			\item 3 < 4 donc $\dfrac{3}{4}$ < 1 ; 15 > 12 donc $\dfrac{15}{12}$ > 1.
			
			\vspace*{0.5cm}		
			
			\item On peut conclure que $\dfrac{3}{4}$ < 1 < $\dfrac{15}{12}$.
		
		\end{itemize}
	}
	
	
	

\end{myex}
