\begin{myprop}
	\begin{itemize}
		\item Dans une expression sans parenthèses, la multiplication est \kw{prioritaire} sur l'addition et la soustraction.
		
		\item Dans une expression avec des parenthèses, on effectue d'abord les calculs à \kw{entre parenthèses}.
	\end{itemize}
\end{myprop}


\begin{myexs}
	Je calcule les expressions suivantes
	
	\begin{multicols}{2}
		\begin{eqnarray*}
			A &=& 2 + 3 \times 4 \\
			A &=& 2 + 12 \\
			A &=& 14
		\end{eqnarray*}
	
	
		\begin{eqnarray*}
			B &=& (2 + 3) \times 4 \\
			B &=& 5 \times 4 \\
			B &=& 20
		\end{eqnarray*}
	\end{multicols}
\end{myexs}
