\section{Vocabulaire (4 points)}

\begin{questions}
	\question[2] \'Ecrire le nombre 15 comme une somme de deux produits.
	\begin{solution}
		\begin{equation*}
			5 \times 2 + 5 \times 1 = 15
		\end{equation*}
	\end{solution}
	
	\question[2] \'Ecrire le nombre 15 comme un produit de deux sommes.
	
	\begin{solution}
		\begin{equation*}
		(1 + 2) \times (3 + 2) = 15
		\end{equation*}
	\end{solution}
	
\end{questions}
