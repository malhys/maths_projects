\section{Vocabulaire  (\`A faire sur cette feuille)}

\begin{questions}
	\question \'Ecrire le nombre 15 comme une somme de deux produits : $ ..........................................= 15$
	\begin{solution}
		\begin{equation*}
			5 \times 2 + 5 \times 1 = 15
		\end{equation*}
	\end{solution}
	\vspace*{0.5cm}
	
	\question \'Ecrire le nombre 15 comme un produit de deux sommes : $..........................................=15$
	
	\begin{solution}
		\begin{equation*}
		(1 + 2) \times (3 + 2) = 15
		\end{equation*}
	\end{solution}
	
\end{questions}
