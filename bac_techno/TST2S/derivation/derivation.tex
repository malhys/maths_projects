	\section{Fonction dérivée}
	
	\begin{mydef}
		Soit une fonction $f$. Pour tout $x$, la \kw{fonction dérivée} $f'$ est la fonction qui donne le nombre dérivé en $x$ (noté \kw{$f'(x)$}). 
		
		Une fonction est \kw{dérivable} en $x$ si et seulement si sa fonction dérivée $f'$ est définie en $x$.
	\end{mydef}
	
	\begin{myex}
		On considère la fonction $f$ définie par 
		\begin{align*}
		f(x) = x^2+2x-1
		\end{align*}
		
		Sa fonction dérivée, est la fonction $f'$ telle que:
		\begin{align*}
			f'(x) = 2x+2
		\end{align*}
	\end{myex}

	\begin{myprop}
		Les variations d'une fonction en un point sont liées au signe du nombre dérivé de la fonction en ce point :
		\begin{itemize}
			\item Si \kw{$f'(x) > 0$}, alors la fonction \kw{$f$ est croissante} en $x$.
			\item Si \kw{$f'(x)=0$}, alors la fonction \kw{$f$ est constante} en $x$.
			\item Si \kw{$f'(x) < 0$}, alors la fonction \kw{$f$ est décroissante} en $x$. 
		\end{itemize}
	\end{myprop}
	
	\begin{myex}
		On considère la fonction $f$, définie sur $[-2;2]$ par $f(x)=-3x^2+6x-4$.
		
		Sa fonction dérivée $f'$ est définie par $f'(x)= -6x +6$.
		
		On résous l'inéquation $f'(x) > 0$.
		
		\begin{align*}
			-6x +6 &> 0 \\
			-6x &> -6 \\
			x &< \frac{-6}{-6} \\
			x &< 1
		\end{align*}
		
		Donc la fonction $f$ est croissante avant $1$ et décroissante après.
