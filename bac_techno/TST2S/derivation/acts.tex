	\begin{myex}
		On considère la fonction $f$, définie sur $[-2;2]$ par $f(x)=-3x^2+6x-4$.
		
		Sa fonction dérivée $f'$ est définie par $f'(x)= -6x +6$.
		
		On résous l'inéquation $f'(x) > 0$.
		
		\begin{align*}
			-6x +6 &> 0 \\
			-6x &> -6 \\
			x &< \frac{-6}{-6} \\
			x &< 1
		\end{align*}
		
		Donc la fonction $f$ est croissante avant $1$ et décroissante après.
		
		On a donc :
		
		\begin{center}
			\begin{tikzpicture}[scale=0.8]
				\tkzTabInit{$x$/1.5,$f'(x)$/1.5,$f(x)$/4}{$-2$, $1$, $2$}
				\tkzTabLine { ,+ ,z , -, }
				\tkzTabVar{-/$-28$,+/$-1$,-/$-4$}
			\end{tikzpicture}
		\end{center}
		
	\end{myex}
