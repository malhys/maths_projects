\begin{questions}
	
	
	\question[] Calculer $P(A) $ et $P(B)$.
	\question[] $\bar{A}$ est l'événement contraire de $A$.
	\begin{parts}
		\part[] Définir $\bar{A}$ par une phrase.
		\part[] Calculer $P(\bar{A})$.
	\end{parts}
	
	\question[] 
	\begin{parts}
		\part[] Définir par une phrase l'événement $A \cap\ B$.
		\part[] Calculer $P(A \cap\ B)$.\label{q:inter}
		\part[] Les événements $A$ et $B$ sont-ils incompatibles.
	\end{parts}
	
	\question[] 
	\begin{parts}
		\part[] Définir par une phrase l'événement $A \cup\ B$.
		\part[] Calculer $P(A \cap\ B)$.\label{q:union}
		\part[] Calculer $P(A) + P(B)$.\label{q:union_b}
		\part[] Retrouver le résultat obtenu au \ref{q:union} à partir des questions précédentes
	\end{parts}
	
	\question[] 
	\begin{parts}
		\part Trouver un événement $C$ tel que $B$ et $C$ soient incompatibles.
		\part Calculer $P(C)$ 
		\part Définir par une phrase l'événement $B \cup\ C$
		\part Calculer $P(B \cup\ C)$
	\end{parts}
	
\end{questions}
