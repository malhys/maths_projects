\begin{eqnarray}
\cos'(a) &=& -\sin(a) \\
\sin'(a) &=& \cos(a)
\end{eqnarray}

\begin{eqnarray}
\int \cos(a) \: da &=& \sin(a)+c \\
\int \sin(a) \: da &=& -\cos(a)+c
\end{eqnarray}

\end{multicols}
