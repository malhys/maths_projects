\begin{eqnarray}
\cos'(A) &=& -\sin(A) \\
\sin'(A) &=& \cos(A)
\end{eqnarray}

\begin{eqnarray}
\int \cos(A)dA &=& \sin(A)+C \\
\int \sin(A)dA &=& -\cos(A)+C
\end{eqnarray}

\end{multicols}
